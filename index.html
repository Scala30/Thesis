üêç Python server setup:
1. Install Python and run: pip install flask flask-cors pillow pyinsane2

from flask import Flask, send_file
from flask_cors import CORS
import pyinsane2
from PIL import Image
import io

app = Flask(__name__)
CORS(app)

def scan_document():
    pyinsane2.init()
    try:
        devices = pyinsane2.get_devices()
        if not devices:
            raise Exception("No scanner found")
        scanner = devices[0]
        scanner.options['resolution'].value = 300
        scanner.options['mode'].value = 'Color'
        scanner.options['source'].value = 'FlatBed'
        scan_session = scanner.scan(multiple=False)
        scan_session.read()
        img = scan_session.images[-1]
        img_byte_arr = io.BytesIO()
        img.save(img_byte_arr, format='JPEG')
        img_byte_arr.seek(0)
        return img_byte_arr
    finally:
        pyinsane2.exit()

@app.route('/scan', methods=['GET'])
def scan():
    try:
        image_data = scan_document()
        return send_file(image_data, mimetype='image/jpeg')
    except Exception as e:
        return str(e), 500

if __name__ == '__main__':
    app.run(host='127.0.0.1', port=5000, debug=False)
